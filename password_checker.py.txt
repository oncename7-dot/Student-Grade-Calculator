attempts = 3

while attempts > 0:
    password = input("Enter password: ")

    has_upper = False
    has_lower = False
    has_digit = False
    has_special = False

    for c in password:
        if c >= 'A' and c <= 'Z':
            has_upper = True
        elif c >= 'a' and c <= 'z':
            has_lower = True
        elif c >= '0' and c <= '9':
            has_digit = True
        else:
            has_special = True

    if len(password) < 8:
        print("Too short")
    elif has_upper == False:
        print("No uppercase letter")
    elif has_lower == False:
        print("No lowercase letter")
    elif has_digit == False:
        print("No number")
    elif has_special == False:
        print("No special character")
    else:
        print("Strong Password")
        break

    attempts = attempts - 1
    print("Attempts left:", attempts)

if attempts == 0:
    print("Too many attempts!")
